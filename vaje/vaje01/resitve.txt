		int stevilo = getchar() - '0';
		getchar();

			prvo = stevilo;

		}
	}
	
	/*if(vsaEnaka) {

#include <stdio.h>

int main() {
	int dolzinaZaporedja = getchar() - '0';
	getchar(); // presledek
	
	int najvecje = -1; //največje število doslej
	int drugoNajvecje = -1; // drugoNajvecjeSteviloDoslej
	
	for(int i = 0; i < dolzinaZaporedja; i++) {
		int stevilo = getchar() - '0';
		getchar();
		
		if(stevilo > najvecje) {
			// <stevilo> je novi prvak in doslej najvecje postane drugo najvecje
			drugoNajvecje = najvecje;
			najvecje = stevilo;
		} else if (stevilo > drugoNajvecje) {
			// <stevilo> je novi "podprvak"
			drugoNajvecje = stevilo;
		}
	}	
	putchar(drugoNajvecje + '0');
	putchar('\n');	
	return 0;
}
